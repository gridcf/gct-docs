223 /*
224  * These are string config options that must be copied between the
225  * Match sub-config and the main config, and must be sent from the
226  * privsep slave to the privsep master. We use a macro to ensure all
227  * the options are copied and the copies are done in the correct order.
228  *
229  * NB. an option must appear in servconf.c:copy_set_server_options() or
230  * COPY_MATCH_STRING_OPTS here but never both.
231  */
232 #define COPY_MATCH_STRING_OPTS() do { \
233  M_CP_STROPT(banner); \
234  M_CP_STROPT(trusted_user_ca_keys); \
235  M_CP_STROPT(revoked_keys_file); \
236  M_CP_STROPT(authorized_keys_command); \
237  M_CP_STROPT(authorized_keys_command_user); \
238  M_CP_STROPT(authorized_principals_file); \
239  M_CP_STROPT(authorized_principals_command); \
240  M_CP_STROPT(authorized_principals_command_user); \
241  M_CP_STROPT(hostbased_key_types); \
242  M_CP_STROPT(pubkey_key_types); \
243  M_CP_STRARRAYOPT(authorized_keys_files, num_authkeys_files); \
244  M_CP_STRARRAYOPT(allow_users, num_allow_users); \
245  M_CP_STRARRAYOPT(deny_users, num_deny_users); \
246  M_CP_STRARRAYOPT(allow_groups, num_allow_groups); \
247  M_CP_STRARRAYOPT(deny_groups, num_deny_groups); \
248  M_CP_STRARRAYOPT(accept_env, num_accept_env); \
249  M_CP_STRARRAYOPT(auth_methods, num_auth_methods); \
250  } while (0)
